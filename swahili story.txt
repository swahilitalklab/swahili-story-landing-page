    // Signup handler — shows inline success and submits to MailerLite in new tab
    function handleSignup(e) {
      const form = document.getElementById('ml-signup-form');
      const success = document.getElementById('form-success');
      const email = form.querySelector('input[type="email"]').value;

      if (!email) return;

      // Track event
      gtag('event', 'signup_submit', { method: 'email' });

      // Show success state after a brief delay (form posts to blank target)
      setTimeout(function() {
        form.style.display = 'none';
        success.style.display = 'block';
      }, 400);
    }
